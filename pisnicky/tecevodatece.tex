\rep{čo som sa naplakal, sivá holubienka.}

\rep{odniesol's mi klúčik od mojej slobody.}

\rep{sloboděnka moja, ta sa nenavrátí.}
}

\rep{sloboděnka moja, ta sa nenavracá.}
